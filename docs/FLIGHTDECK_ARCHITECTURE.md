## Executive Summary

The FlightDeck System is a comprehensive dashboard ecosystem providing real-time visibility, control, and analytics across all TSI Enterprise applications. The system consists of **12 Individual App FlightDecks** unified under a single **Alpha FlightDeck** command center.

### Core Principles

1. **Single Source of Truth** - Alpha FlightDeck aggregates all metrics
2. **Real-time Updates** - WebSocket connections for live data
3. **Role-Based Access** - Granular permissions per app
4. **Offline Resilient** - Service workers for offline access
5. **Mobile First** - Responsive design for all devices

---

## Alpha FlightDeck (Master Dashboard)

### Purpose
The Alpha FlightDeck is the central command center providing:
- Portfolio-wide visibility across all 12 applications
- Aggregated KPIs and health metrics
- Quick navigation to individual FlightDecks
- Cross-project resource allocation
- Enterprise-level alerts and notifications

### GitHub Integration

```typescript
// GitHub Webhook Handler
interface GitHubWebhookPayload {
  action: string;
  repository: {
    name: string;
    full_name: string;
  };
  commits?: GitHubCommit[];
  pull_request?: GitHubPR;
  release?: GitHubRelease;
}

// Sync GitHub Data
async function syncGitHubMetrics(app: string): Promise<GitHubMetrics> {
  const gh = new GitHubClient(process.env.GITHUB_TOKEN);

  const [commits, prs, issues] = await Promise.all([
    gh.getRecentCommits(`TSIKanati/${app}`, 7),
    gh.getOpenPRs(`TSIKanati/${app}`),
    gh.getOpenIssues(`TSIKanati/${app}`)
  ]);

  return {
    commitsThisWeek: commits.length,
    openPRs: prs.length,
    openIssues: issues.length,
    lastCommit: commits[0]?.date
  };
}
```

## Database Schema

### Core Tables

```sql
-- FlightDeck Core Schema

-- Users & Auth
CREATE TABLE fd_users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(50) NOT NULL DEFAULT 'viewer',
    mfa_enabled BOOLEAN DEFAULT false,
    mfa_secret VARCHAR(255),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    last_login TIMESTAMPTZ
);

-- App Registry
CREATE TABLE fd_apps (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    icon VARCHAR(10),
    priority VARCHAR(10) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'development',
    github_repo VARCHAR(255),
    local_path VARCHAR(500),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- App Metrics (Time Series)
CREATE TABLE fd_metrics (
    id BIGSERIAL PRIMARY KEY,
    app_id VARCHAR(50) REFERENCES fd_apps(id),
    metric_name VARCHAR(100) NOT NULL,
    metric_value DECIMAL(20, 4) NOT NULL,
    tags JSONB DEFAULT '{}',
    timestamp TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_metrics_app_time ON fd_metrics(app_id, timestamp DESC);

-- Alerts
CREATE TABLE fd_alerts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    app_id VARCHAR(50) REFERENCES fd_apps(id),
    severity VARCHAR(20) NOT NULL,
    message TEXT NOT NULL,
    details JSONB DEFAULT '{}',
    acknowledged BOOLEAN DEFAULT false,
    acknowledged_by UUID REFERENCES fd_users(id),
    acknowledged_at TIMESTAMPTZ,
    resolved BOOLEAN DEFAULT false,
    resolved_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- User Permissions
CREATE TABLE fd_permissions (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES fd_users(id),
    permission VARCHAR(100) NOT NULL,
    granted_by UUID REFERENCES fd_users(id),
    granted_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, permission)
);

-- Audit Log
CREATE TABLE fd_audit_log (
    id BIGSERIAL PRIMARY KEY,
    user_id UUID REFERENCES fd_users(id),
    action VARCHAR(100) NOT NULL,
    app_id VARCHAR(50),
    details JSONB DEFAULT '{}',
    ip_address INET,
    user_agent TEXT,
    result VARCHAR(20),
    timestamp TIMESTAMPTZ DEFAULT NOW()
);
CREATE INDEX idx_audit_user ON fd_audit_log(user_id, timestamp DESC);

-- Dashboard Configurations
CREATE TABLE fd_dashboard_config (
    id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES fd_users(id),
    app_id VARCHAR(50) REFERENCES fd_apps(id),
    widget_layout JSONB NOT NULL,
    preferences JSONB DEFAULT '{}',
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, app_id)
);

-- Deployments
CREATE TABLE fd_deployments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    app_id VARCHAR(50) REFERENCES fd_apps(id),
    triggered_by UUID REFERENCES fd_users(id),
    branch VARCHAR(100),
    commit_sha VARCHAR(40),
    status VARCHAR(20) NOT NULL DEFAULT 'pending',
    started_at TIMESTAMPTZ DEFAULT NOW(),
    completed_at TIMESTAMPTZ,
    logs TEXT
);
```
